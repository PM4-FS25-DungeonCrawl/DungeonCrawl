70void game_loop() {
71 bool running = true;//should only be set in the state machine
72
73 while (running) {
74 // Process current game state
75 switch (current_state) {
76 case MAIN_MENU:
77 main_menu_state();
78 break;
79
80 case GENERATE_MAP:
81 generate_map();
82 current_state = MAP_MODE;
83 break;
84
85 case MAP_MODE:
86 map_mode_state();
87 break;
88
89 case COMBAT_MODE:
90 combat_mode_state();
91 break;
92
93 case LOOT_MODE:
94 loot_mode_state();
95 break;
96
97 case INVENTORY_MODE:
98 inventory_mode_state();
99 break;
100
101 case STATS_MODE:
102 stats_mode_state();
103 break;
104
105 case EXIT:
106 running = false;
107 break;
108 }
109 }
110
111 // Close database connection
112 db_close(&db_connection);
113}
